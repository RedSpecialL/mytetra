void ProcessAsUser(std::wstring& args)

{

DWORD sessionId = ::WTSGetActiveConsoleSessionId();

CHECK_OS_ERROR(sessionId != 0xFFFFFFFF);


HANDLE userToken;

CHECK_OS_ERROR(::WTSQueryUserToken(sessionId, &userToken));


STARTUPINFO startupInfo;

::ZeroMemory(&startupInfo, sizeof(STARTUPINFO));

startupInfo.cb = sizeof(STARTUPINFO);


PROCESS_INFORMATION processInfo;

::ZeroMemory(&processInfo, sizeof(PROCESS_INFORMATION));


CHECK_OS_ERROR(::CreateProcessAsUser(

userToken,

nullptr,

args.empty() ? nullptr : &args[0],

nullptr,

nullptr,

FALSE,

NORMAL_PRIORITY_CLASS,

nullptr,

nullptr,

&startupInfo,

&processInfo));


CHECK_OS_ERROR(::WaitForSingleObject(processInfo.hProcess, INFINITE) == WAIT_OBJECT_0);


//DWORD exitCode;

//CHECK_OS_ERROR(::GetExitCodeProcess(processInfo.hProcess, &exitCode));

//if (exitCode != STILL_ACTIVE)

// LDBG_ << "Process exit with code: " << exitCode;


::CloseHandle(userToken);

::CloseHandle(processInfo.hProcess);

::CloseHandle(processInfo.hThread);

}
