bool IsProcessRun()

{

LDBG_ << "Trying to find is tray process running.";

HANDLE hSnap = NULL;

PROCESSENTRY32 pe32;

hSnap = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);

if (hSnap != NULL)

{

if (Process32First(hSnap, &pe32))

{

if (StrCmpW(pe32.szExeFile, L"QuicktechTrayAgent.exe") == 0)

{

return TRUE;

}

while (Process32Next(hSnap, &pe32))

{

if (StrCmpW(pe32.szExeFile, L"QuicktechTrayAgent.exe") == 0)

{

return TRUE;

}

}

}

}

CloseHandle(hSnap);

return FALSE;

}
